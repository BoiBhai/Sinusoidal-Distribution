\begin{frame}{Sinusoidal Distribution}
    $X \follows \Sinu(\alpha, \delta, \omega, \chi)$:
    $$f_X (x) = \frac{1}{A} \sin^\chi \pi \left( \frac{x-\alpha}{\delta} \right)^\omega$$
\begin{itemize}
    \item Location parameter $\alpha \in \RR$
    \item Scale parameter $\delta>0$
    \item Asymmtery parameter $\omega \geq 0$
    \item Peakedness parameter $\chi \geq 0$
    \item $A$ is the area under $\sin^\chi {\pi (\frac{x-\alpha}{\delta})^\omega}$
\end{itemize}
\end{frame}
